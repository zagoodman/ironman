# despite having unique bib_n, some athleteids have duplicate entries. 
# Why? Because:
# 1. athleteid is assigned at reg, and some family members reg together.
# 2. athletes assigned multiple bib_n and only counted once
# 3. some athletes listed in multiple categories

# Goal: assign unique athleteid such that raceid-athleteid uniquely identifies observations
# To do this, we will add 1e7 to athleteids that are redundant
# Importantly, we want the true account holder to stay with the original athleteid
# We will match the true account holder using the name on the account or gender
# If there are different divisions but same name/gender, we will give the older priority


# Steps:
# 1) identify accounts that need spliting
# 2) separate further accounts that can be split 1) by gender, 2) by name and not gender, 3) by division and not name or gender
# 3) among accounts that cannot be split (constant in race-id-name-gender-division), assign the faster account priority

# Notes: for 2), change the athleteid for those in the main df, in case some "guests" raced solo, but only for gender and name
#  For 2.3 and 3, do NOT apply changes to main DF as later we will look for age group issues.
# Some entries are errors (e.g. pros listed as amateurs), but in these steps we will not delete entries.

# name cleaning

# If comma, move everything after the comma to front of name
# lowercase everything
# remove special characters

def name_clean(n):
    """
    Takes a name of the form Last, First Middle
    And returns first middle last. Note that Last can
    have multiple parts, and there may not be a middle name.
    Returned name is lowercase.
    Removes common endings to names (like Jr, Sr, etc.)
    """
    # check if string
    if str(n) != n:
        return
    
    # lowercase
    n = n.lower()
    
    # remove prefixes and suffixes
    remove_words = [' jr.,', ' jr.', ' jr', ' dr.,', ' dr.', ' dr', ' sr.', ' sr']
    for j in remove_words:
        n = n.replace(j, "")
    
    # remove special characters (keep only letters)
    n = ''.join(e for e in n if e.isalpha() or e.isspace() or e == ",")
    
    # rearrange if any commas found
    n_list = n.split(", ")
    
    # if no commas found, return n
    if len(n_list) == 1:
        return n
    
    # If comma found, flip order
    name = n_list[1] + " " + n_list[0]
    return name
